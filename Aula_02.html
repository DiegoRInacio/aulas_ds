Estruturas de Dados em Python

Estruturas
# ── LISTA (mutável, ordenada) ──
vendas = [1200, 850, 2300, 970]
vendas.append(1500)
print(vendas[0])       # 1200
print(vendas[-1])      # 1500 (último)
print(vendas[1:3])     # [850, 2300]

# ── DICIONÁRIO (chave: valor) ──
cliente = {
    "nome":   "João",
    "idade":  35,
    "cidade": "São Paulo",
    "ativo":  True
}
print(cliente["nome"])
cliente["score"] = 720

# ── TUPLA (imutável) ──
coord = (-23.55, -46.63)  # lat, lon SP

# ── SET (valores únicos) ──
cats = {"A", "B", "A", "C"}
# resultado: {'A', 'B', 'C'}

📋 Quando usar cada uma?

  • Lista — sequências de valores, dados de linhas
  • Dicionário — registros com atributos (→ linhas de DataFrame)
  • Tupla — valores fixos, coordenadas, retornos de função
  • Set — verificar unicidade, deduplicação
# Lista de dicts → base de dados
dados = [
    {"produto": "Notebook", "valor": 3200},
    {"produto": "Mouse",    "valor": 120},
    {"produto": "Teclado",  "valor": 280},
]
# Isso vira um DataFrame Pandas!
import pandas as pd
df = pd.DataFrame(dados)

Controle de Fluxo e Funções

Lógica
# ── CONDICIONAIS ──
receita = 85000

if receita > 100000:
    print("Meta atingida!")
elif receita > 70000:
    print("Acima da média")
else:
    print("Abaixo da meta")

# ── LOOP FOR ──
meses  = ["Jan", "Fev", "Mar"]
vendas = [1200, 980, 1540]

for mes, v in zip(meses, vendas):
    print(f"{mes}: R$ {v:,.0f}")

# ── LIST COMPREHENSION ──
dobro = [x * 2 for x in vendas]
altos = [v for v in vendas if v > 1000]
# ── FUNÇÕES ──
def calcular_crescimento(atual, anterior):
    """Crescimento percentual entre períodos."""
    if anterior == 0:
        return None
    return ((atual - anterior) / anterior) * 100

cresc = calcular_crescimento(1540, 1200)
print(f"Crescimento: {cresc:.1f}%")
# Crescimento: 28.3%

# ── LAMBDA ──
formatar = lambda x: f"R$ {x:,.2f}"
print(formatar(3200))  # R$ 3.200,00

NumPy — Computação Numérica

NumPy
import numpy as np

# ── Criando arrays ──
vendas = np.array([1200, 850, 2300, 970, 1540])
zeros  = np.zeros(5)
seq    = np.arange(0, 100, 10)

# ── Estatísticas básicas ──
print(np.mean(vendas))     # média: 1372.0
print(np.median(vendas))   # mediana: 1200.0
print(np.std(vendas))      # desvio padrão
print(np.max(vendas))      # 2300

# ── Operações vetorizadas ──
comissao = vendas * 0.05
meta     = vendas > 1000

# ── Array 2D (matriz) ──
matriz = np.array([
    [10, 20, 30],
    [40, 50, 60]
])
print(matriz.shape)  # (2, 3)

⚡ Por que não usar listas Python?

  • Arrays NumPy são até 50x mais rápidos
  • Operações vetorizadas — sem loops manuais
  • Menor consumo de memória
  • Base de Pandas, Scikit-learn, TensorFlow
import time

n = 1_000_000
lista = list(range(n))
arr   = np.arange(n)

# Lista Python: ~0.1s
t0 = time.time()
res = [x*2 for x in lista]
print(f"Lista: {time.time()-t0:.4f}s")

# NumPy: ~0.002s
t0 = time.time()
res = arr * 2
print(f"NumPy: {time.time()-t0:.4f}s")

Pandas — Seleção, Filtro e Agrupamento

Pandas
# ── Selecionar colunas ──
df["preco"]
df[["produto", "preco"]]

# ── Filtrar linhas ──
caros  = df[df["preco"] > 1000]
combo  = df[(df["preco"] > 200) & (df["estoque"] > 20)]

# ── Criar novas colunas ──
df["valor_total"]  = df["preco"] * df["estoque"]
df["desconto_10"] = df["preco"] * 0.9

# ── Agrupar (GROUP BY) ──
resumo = df.groupby("categoria").agg({
    "preco":   ["mean", "max"],

})

# ── Ordenar e tratar nulos ──
df.sort_values("preco", ascending=False)
df.isnull().sum()
df.fillna(0)

🗂️ loc vs iloc

# loc → por RÓTULO (nome)
df.loc[0, "produto"]
df.loc[1:3, "preco"]

# iloc → por POSIÇÃO (índice)
df.iloc[0, 2]
df.iloc[:2, :3]

Visualização — Matplotlib & Seaborn

Visualização
import matplotlib.pyplot as plt
import seaborn as sns

# ── Gráfico de barras ──
fig, ax = plt.subplots(figsize=(8, 4))
ax.bar(df["produto"], df["preco"],
       color="#3776ab")
ax.set_title("Preço por Produto")
ax.set_ylabel("Preço (R$)")
plt.tight_layout()
plt.show()

# ── Histograma com Seaborn ──
sns.histplot(df["preco"], bins=10,
             kde=True, color="#2d9d78")
plt.show()

# ── Heatmap de correlação ──
corr = df.select_dtypes("number").corr()
sns.heatmap(corr, annot=True,
            cmap="coolwarm", fmt=".2f")
plt.show()

📊 Quando usar cada gráfico?

  • Barras — comparar categorias
  • Linha — séries temporais, evolução
  • Histograma — distribuição de uma variável
  • Scatter — relação entre duas variáveis numéricas
  • Boxplot — distribuição + outliers por categoria
  • Heatmap — correlação entre variáveis
  • Pizza — proporção (use com moderação)
# boxplot por categoria
sns.boxplot(data=df,
            x="categoria",
            y="preco",
            palette="Set2")

# scatter com regressão
sns.regplot(data=df,
            x="preco",
            y="estoque")

🏪 Cenário: Análise de Vendas de Supermercado

Dataset CSV com 1.000 transações. Use Python + Pandas + Matplotlib para explorar os dados.

  1. Carregue o arquivo supermarket_sales.csv com Pandas e exiba as 5 primeiras linhas

Guia de Resolução

Referência
import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

# 1. Carregar e inspecionar
df = pd.read_csv("supermarket_sales.csv")
print(df.head())
print(df.info())
print(df.isnull().sum())

# 2. Vendas por filial
por_filial = df.groupby("Branch")["Total"].sum()
print(por_filial)

# 3. Ticket médio por produto
ticket = df.groupby("Product line")["Total"].mean()
print(ticket.sort_values(ascending=False))
# 4. Barras — faturamento por filial
fig, ax = plt.subplots(figsize=(7, 4))
por_filial.plot(kind="bar", ax=ax,
    color=["#3776ab", "#ffd43b", "#2d9d78"],
    edgecolor="none")
ax.set_title("Faturamento por Filial")
ax.tick_params(axis="x", rotation=0)
plt.tight_layout()
plt.show()

# 5. Histograma de avaliações
sns.histplot(df["Rating"], bins=20,
            kde=True, color="#8e44ad")
plt.title("Distribuição das Avaliações")
plt.show()

# 6. Scatter: Total vs Rating
sns.scatterplot(data=df,
    x="Total", y="Rating",
    hue="Branch", alpha=0.6)
plt.show()
